    Example 5: Structured environments and TCL

    This example provides the necessary code to reproduce the calculations from example 4 in 2, where the Hamiltonian is given by

    \begin{align} H = \omega_{0} \sigma_{+} \sigma_{-} + \sum_{k} \omega_{k} b_{k}^{\dagger}b_{k} + \sigma_{+} \otimes \sum_{k} g_{k} b_{k} +H.C . \end{align} For TCL and the RWA exact solution and \begin{align} H = \omega_{0} \sigma_{+} \sigma_{-} + \sum_{k} \omega_{k} b_{k}^{\dagger}b_{k} + \sigma_{x} \sum_{k} g_{k} (b_{k} +b_{k}^{\dagger}) \end{align} For the methods supported by this package and QutiP. The main pourpose of this example is to show the usage of Non-Markovian master equations with structured environments as well as to show how the exponent approximation can be used with TCL and Volterra equations.     def solve_c1_system(cks, vks, c1_0, t_span=(0, 50), t_eval=None):
    """
    Solves the system:
        dot{c1} = -sum yk
        dot{yk} = -vk * yk + ck * c1
    Parameters:
        cks : list or array of c_k coefficients
        vks : list or array of v_k (decay rates)
        c1_0 : float, initial condition for c1(t)
        t_span : tuple, (t0, tf)
        t_eval : array, optional time grid for evaluation
    Returns:
        t, c1, ys : time array, c1(t) array, y_k(t) array
    """
    cks = np.array(cks, dtype=complex)
    vks = np.array(vks, dtype=complex)
    N = len(cks)

    def system(t, u):
        c1 = u[0]
        ys = u[1:]

        # Calculate the time derivatives of c1 and yk
        dc1_dt = -np.sum(ys)
        dys_dt = -(vks) * ys + cks * c1

        # Ensure ys don't go below 0 (clipping to prevent negative values)
        return np.concatenate([[dc1_dt], dys_dt])

    # Initial conditions
    u0 = np.concatenate([[c1_0], np.zeros_like(cks)])

    # Time grid
    if t_eval is None:
        t_eval = np.linspace(t_span[0], t_span[1], 1000)

    # Solve ODE system using 'RK45' method for better stability
    sol = solve_ivp(system, t_span, u0, t_eval=t_eval, method="RK45", vectorized=False)

    t = sol.t
    c1 = sol.y[0]
    ys = sol.y[1:]
    return t, c1, ys


def g4(ck, vk, w0, t):
    """
    Fourth order coefficients for the TCL equation
    Parameters:
        ck : list or array of c_k coefficients
        vk : list or array of v_k (decay rates)
        t : np.array, the times in which to evaluate the coefficients
    Returns:
         np.array The coefficient evaluated at times t
    """
    diff = []
    sam = []
    for i in range(len(ck)):
        for j in range(len(ck)):
            if i != j:
                a = 1j * w0 - vk[i]
                b = vk[j] - 1j * w0
                mul = (ck[i] * ck[j]) / (a * b)
                uno = np.exp(-t * b) / (vk[i] - vk[j])
                dos = np.exp(-t * (vk[i] + vk[j] - 2 * 1j * w0)) / b
                tres = -1 / a
                cuatro = (t - 1 / a - 1 / b + 1 / (vk[j] - vk[i])) * np.exp(t * a)
                diff.append(mul * (uno + dos + tres + cuatro))
            else:
                a = vk[i] - 1j * w0
                mul = (ck[i] ** 2) / (a**3)
                uno = 1 - np.exp(-2 * t * a) - 2 * t * np.exp(-t * a) * a
                sam.append(mul * uno)
    different = sum(diff)
    same = sum(sam)
    return 2 * (different + same)


def g2(ck, vk, w0, t):
    """
    Second order coefficients for the TCL equation
    Parameters:
        ck : list or array of c_k coefficients
        vk : list or array of v_k (decay rates)
        t : np.array, the times in which to evaluate the coefficients
    Returns:
         np.array The coefficient evaluated at times t
    """
    exp = []
    for i in range(len(ck)):
        mul = ck[i] / (vk[i] - 1j * w0)
        fd = 1 - np.exp(-(vk[i] - 1j * w0) * t)
        exp.append(mul * fd)
    return 2 * sum(exp)


def gamma4(ck, vk, w0, t):
    "Fourth order contribution"
    return g4(ck, vk, w0, t) + g2(ck, vk, w0, t)


def master_eq_gen(func, ck, vk, w0, t):
    """
    Exact master equation generator
    Parameters:
        func : python function that generates the coefficients
        ck : list or array of c_k coefficients
        vk : list or array of v_k (decay rates)
        w0 : qubit frequency
        t : np.array, the times in which to evaluate the coefficients
    Returns:
         np.array, The coefficient evaluated at times t
    """
    gamma = func(ck, vk, w0, t).real
    s = -1j * func(ck, vk, w0, t).imag / 2
    A = sigmap() * sigmam()
    comm = spre(A) - spost(A)
    relax = spre(sigmam()) * spost(sigmap())
    relax2 = -(spre(A) + spost(A)) / 2
    return s * comm + gamma * (relax + relax2)
